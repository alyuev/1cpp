//======================================================================
Процедура ЗаписатьОбразОбъекта(ТипОбъектаИлиОбъектКласса) Экспорт
	Перем ТабМетоды, ТабСвойства;

	Конт(Контекст).ПолучитьДанныеОТипе(ТипОбъектаИлиОбъектКласса, ТабМетоды, ТабСвойства);
	
	лСписок = СоздатьОбъект("СписокЗначений");
	лСписок.Установить("Методы", 	ТабМетоды);
	лСписок.Установить("Свойства",	ТабСвойства);
	
	Если ТипЗначенияСтр(ТипОбъектаИлиОбъектКласса) = "Строка" Тогда
		ЗначениеВФайл(КаталогОбразов+ТипОбъектаИлиОбъектКласса+".lst", лСписок, 1);
	Иначе
		ЗначениеВФайл(КаталогОбразов+ТипЗначенияСтр(ТипОбъектаИлиОбъектКласса)+".lst", лСписок, 1);
	КонецЕсли;	
КонецПроцедуры // ЗаписатьОбразОбъекта() 


//*************************************************************************
Процедура ВосстановитьОбразОбъекта(ТипОбъекта, ТабМетоды, ТабСвойства) Экспорт
	Перем лСписок;

	лСписок = "";
	Если ФС.СуществуетФайл(КаталогОбразов+ТипОбъекта+".lst") = 0 Тогда
		Модуль = ВМ();               
		лСтрокаИсключения = "Нет файла образа для объекта """+ТипОбъекта+"""";
		Модуль.ВыброситьИскл(лСтрокаИсключения, лСтрокаИсключения);
	КонецЕсли;
	
	ЗначениеИзФайла(КаталогОбразов+ТипОбъекта+".lst", лСписок, 1);
	
	Если ТипЗначенияСтр(лСписок) <> "СписокЗначений" Тогда
		Модуль = ВМ();               
		лСтрокаИсключения = "Неверный образ для объекта '"+ТипОбъекта+"'";
		Модуль.ВыброситьИскл(лСтрокаИсключения, лСтрокаИсключения);
	КонецЕсли;
	
	ТабМетоды = лСписок.Получить("Методы");
	ТабСвойства = лСписок.Получить("Свойства");
	
	Если ТипЗначенияСтр(ТабМетоды) <> "ТаблицаЗначений" Тогда
		Модуль = ВМ();               
		лСтрокаИсключения = "Неверный образ для объекта '"+ТипОбъекта+"' - неверная таблица методов";
		Модуль.ВыброситьИскл(лСтрокаИсключения, лСтрокаИсключения);
	КонецЕсли;
	
	Если ТипЗначенияСтр(ТабСвойства) <> "ТаблицаЗначений" Тогда
		Модуль = ВМ();               
		лСтрокаИсключения = "Неверный образ для объекта '"+ТипОбъекта+"' - неверная таблица свойств";
		Модуль.ВыброситьИскл(лСтрокаИсключения, лСтрокаИсключения);
	КонецЕсли;
	
КонецПроцедуры //ПолучитьДанныеОТипе
