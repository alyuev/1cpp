Процедура Тест_СоздатьЭлементУправленияОшибка() Экспорт
	Сам = Сам();
	
	Конт = ОткрытьФормуЭлементаУправления();

	Сам.ПроверитьИсключение(Форма, "СоздатьЭлементУправления", "НеважноЧто", "текстАктивИкс");
	
	Форма.УстановитьФорму(Конт.Форма);  
	Сам.ПроверитьИсключение(Форма, "СоздатьЭлементУправления", "ПростоСтрока", "текстАктивИкс");
	Сам.ПроверитьИсключение(Форма, "СоздатьЭлементУправления", "ООППотомок", "текстАктивИкс");
	
	ОбъектНеПотомокЭлементУправления = СоздатьОбъект("ООППредок");
	Сам.ПроверитьИсключение(Форма, "СоздатьЭлементУправления", ОбъектНеПотомокЭлементУправления, "текстАктивИкс");

	ЭлементУправления = ЭлементУправления();
	Сам.ПроверитьИсключение(Форма, "СоздатьЭлементУправления", ЭлементУправления, "НесуществующийРеквизит");

Процедура СоздатьЭлементУправленияЧерезАтрибутРасш(ФлагАтрибута, ФлагПростойВызов)
	Сам = Сам();
	
	Конт = ОткрытьФормуЭлементаУправления();
	                      
	Если ФлагАтрибута = 1 Тогда
		АтрибутИлиИДАтрибута = Конт.Форма.текстАктивИкс;
	Иначе
		АтрибутИлиИДАтрибута = "текстАктивИкс";
	КонецЕсли;

	ЭлементУправления = ЭлементУправления();
	Если ФлагПростойВызов = 1 Тогда
		Форма.УстановитьФорму(Конт.Форма);  
		СозданныйЭлементУправления = Форма.СоздатьЭлементУправления(ЭлементУправления, АтрибутИлиИДАтрибута);
	Иначе
		СозданныйЭлементУправления = Форма.СоздатьЭлементУправленияНаФорме(Конт.Форма, ЭлементУправления, АтрибутИлиИДАтрибута);
	КонецЕсли;

	Сам.ПроверитьРавенство(ТипЗначенияСтр(Форма), ИмяКласса);
	
КонецПроцедуры	//

Процедура Тест_СоздатьЭлементУправленияНаФормеОшибка() Экспорт
	Сам = Сам();
	
	Конт = ОткрытьФормуЭлементаУправления();

	Сам.ПроверитьИсключение(Форма, "СоздатьЭлементУправленияНаФорме", Конт.Форма, "НеважноЧто", "текстАктивИкс");
	
	Форма.УстановитьФорму(Конт.Форма);  
	Сам.ПроверитьИсключение(Форма, "СоздатьЭлементУправленияНаФорме", Конт.Форма, "ПростоСтрока", "текстАктивИкс");
	Сам.ПроверитьИсключение(Форма, "СоздатьЭлементУправленияНаФорме", Конт.Форма, "ООППотомок", "текстАктивИкс");
	
	ОбъектНеПотомокЭлементУправления = СоздатьОбъект("ООППредок");
	Сам.ПроверитьИсключение(Форма, "СоздатьЭлементУправленияНаФорме", Конт.Форма, ОбъектНеПотомокЭлементУправления, "текстАктивИкс");

	ЭлементУправления = ЭлементУправления();
	Сам.ПроверитьИсключение(Форма, "СоздатьЭлементУправленияНаФорме", Конт.Форма, ЭлементУправления, "НесуществующийРеквизит");

КонецПроцедуры	// 

Процедура СоздатьЭлементУправленияНаФормеСВызовомСобытия(Событие, СтрокаЖурнала)
	Сам = Сам();
	                     
	Конт = ОткрытьФормуЭлементаУправления(Событие);

	Журнал = Конт.Форма.Параметр; // TurboBL
	Сам.ПроверитьРавенство(ТипЗначенияСтр(Журнал), "ЖурналВызоваМетодов");
	
	Форма.УстановитьФорму(Конт.Форма);  
	
	ЭлементУправления = ЭлементУправления();
	СозданныйЭлементУправления = Форма.СоздатьЭлементУправленияНаФорме(Конт.Форма, ЭлементУправления, "текстАктивИкс");

	Сам.ПроверитьРавенство(Журнал.стрЖурнал, СтрокаЖурнала);
	
КонецПроцедуры	//
