Функция СобытиеГМ_ВводНового(КонтФормы, ФлагКопирования, ОбъектКопирования) Экспорт
	Журнал.ДобавитьЗапись("КОП::СобытиеГМ_ВводНового");
	Если ТипЗначения(ФлагКопирования) <> 0  Тогда
		Журнал.ДобавитьЗапись("ФлагКопирования::"+ФлагКопирования);
	КонецЕсли;
	Если ТипЗначения(ОбъектКопирования) <> 0  Тогда
		Если ПустоеЗначение((ОбъектКопирования)) <> 1  Тогда
			Журнал.ДобавитьЗапись("ОбъектКопирования::"+ТипЗначенияСтр(ОбъектКопирования));
		Иначе
			Журнал.ДобавитьЗапись("ОбъектКопирования::ПустоеЗначение_"+ТипЗначенияСтр(ОбъектКопирования));
		КонецЕсли;
	КонецЕсли;
	
	Возврат Флаг_СобытиеГМ_ВводНового_ВозвращаемоеЗначение;
