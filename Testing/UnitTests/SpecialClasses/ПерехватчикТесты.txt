Процедура Тест_ПерехватСобытийПроведенияДокумента() Экспорт
	Сам = Сам();

	Журнал=СоздатьОбъект("ЖурналВызоваМетодов");

	Док = СоздатьДок();
	ПредставлениеДокумента = _ВидДокументаДляПерехватаСобытийПроведения+" № "+СокрЛП(Док.НомерДок)+" от "+Док.ДатаДок;

	лОбработчик = Перехватчик.ПолучитьПерехватчикСобытийПроведенияДокумента(_ВидДокументаДляПерехватаСобытийПроведения);

	// а нужно, чтобы он сработал сразу же
	// одна копия объекта класса хранится внутри перехватчика
	КлассПерехватчикГК = 0;

	ДОк.Провести();

	//Сообщить("Журнал.стрЖурнал = <"+Журнал.стрЖурнал+">");
	Сам.ПроверитьРавенство(Журнал.стрЖурнал, "КОП::Инициализация, КОП::ОбработкаПроведения<"+ПредставлениеДокумента+">0, Модуль::ОбработкаПроведения<"+ПредставлениеДокумента+">");
	
	стрСохр = Журнал.стрЖурнал;
	
	Перехватчик.СнятьПерехватСобытийПроведенияДокумента(_ВидДокументаДляПерехватаСобытийПроведения);
	Сам.ПроверитьРавенство(Журнал.стрЖурнал, стрСохр+", КОП::Деструктор");

	лОбработчик = Перехватчик.ПолучитьПерехватчикСобытийПроведенияДокумента(_ВидДокументаДляПерехватаСобытийПроведения);
	Сам.ПроверитьТипЗначения(лОбработчик, "");
