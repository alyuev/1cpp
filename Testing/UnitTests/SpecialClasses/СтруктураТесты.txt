Процедура Тест_ЦиклПоСтруктуреСоВставкой() Экспорт

	Структура = Структура();
	
	Структура.Вставить("Ключ1", 0);
	Структура.Вставить("Ключ2", 0);
	
	Структура.Ключ1=1;
	Структура.Ключ2=1;

	ОчиститьСтруктуру(Структура);

	Структура.Ключ1=Структура.Ключ1+1;
	Структура.Ключ2=Структура.Ключ2+1;
	
	Сам.ПроверитьРавенство(Структура.Ключ1, 1);
	Сам.ПроверитьРавенство(Структура.Ключ2, 1);
