Процедура ДобавитьПадение(ОписаниеТеста, Ошибка, ОбъектИсключения) Экспорт
	_ВсегоТестов = _ВсегоТестов + 1;
	Если ТипЗначенияСтр(ОбъектИсключения)="Исключение_ПропуститьТест" Тогда
		_ПропущеноТестов = _ПропущеноТестов + 1;
		стрОписание=стрСоединить(ОписаниеТеста, ОбъектИсключения.Причина, РазделительСтрок+"       Причина: ");
		_ПропущенныеТесты.ДобавитьЗначение(стрОписание);
		Возврат;
	КонецЕсли;
	_УпавшихТестов = _УпавшихТестов + 1;
	Описание = Ошибка + РазделительСтрок + ОбъектИсключения;
	Если (СокрЛП(Ошибка) = "") и ((ПустоеЗначение(ОбъектИсключения) = 1)) Тогда
		_Ошибки.ДобавитьЗначение("Неизвестная ошибка", ОписаниеТеста); // иначе число ошибок не будет совпадать с числом упавших тестов
	Иначе
		_Ошибки.ДобавитьЗначение(Описание, ОписаниеТеста);
	КонецЕсли;
	_ВремяЗавершения=_GetPerformanceCounter();
КонецПроцедуры

Процедура Печать() Экспорт
	Перем ОписаниеТеста;
	
	Если _фСообщатьТолькоВЛогФайл = 0 Тогда
		Трассировка = СоздатьОбъект("Общие.Трассировка");
		Трассировка.СоздатьТрассировку("Тестирование 1С++");
	КонецЕсли;
                                   
	Если ВсегоТестов() > 0 Тогда
		Сообщить(Итог(), ?(УпавшихТестов()>0, "!!!", "I"));
        Сообщить("Общее время выполнения: "+Формат(ВремяВыполнения()/1000, "Ч10.2")+" сек.", ".");
	КонецЕсли;

	Для ъ=1 По УпавшихТестов() Цикл
		Ошибка = _Ошибки.ПолучитьЗначение(ъ, ОписаниеТеста);
		Сообщить("");
		Сообщить(""+ъ+". Упал тест: "+ОписаниеТеста, "!!!");
		Сообщить(Ошибка);
	КонецЦикла;

	Если ПропущеноТестов()>0 Тогда
		Сообщить("");
		Сообщить("Пропущены тесты: ", "I");
		Для ъ=1 По ПропущеноТестов() Цикл
			Сообщить(""+ъ+". "+_ПропущенныеТесты.ПолучитьЗначение(ъ), "i");

		Трассировка.ПоказатьТрассировку();
	КонецЕсли;

	Если фОчиститьЛог = 1 Тогда

			ФС.УдалитьФайл(ИмяЛогФайла);
		КонецЕсли;
	КонецЕсли;
	
	Если УпавшихТестов() = 0 Тогда

		ЛогФайл1.Сообщить("Пропущены тесты: ", "!!!");

			ЛогФайл1.Сообщить(""+ъ+". "+_ПропущенныеТесты.ПолучитьЗначение(ъ), "i");

		ЛогФайл1.Сообщить("");
		ЛогФайл1.Сообщить(""+ъ+". Упал тест: "+ОписаниеТеста, "!!!");

	Если ВсегоТестов() > 0 Тогда
