функция ПолучитьСписокВсехКлассовЮнитТестовЗагруженныхВ1С(ПрефиксИлиСуффиксИмениКласса) экспорт
	Список = СоздатьОбъект("СписокЗначений");
	набор = СоздатьОбъект("НаборТестов");

	MetaInfoClasses = СоздатьОбъект("MetaInfoClasses");
	СписокВсехКлассов1С = MetaInfoClasses.ПолучитьВсеИменаКлассов();

	Длина = СтрДлина(ПрефиксИлиСуффиксИмениКласса);
	ПрефиксИлиСуффикс1 = Врег(ПрефиксИлиСуффиксИмениКласса);

	для ъ = 1 по СписокВсехКлассов1С.РазмерСписка() цикл
		ИмяКласса = СписокВсехКлассов1С.ПолучитьЗначение(ъ);
		ИмяКлассаВрег = Врег(СписокВсехКлассов1С.ПолучитьЗначение(ъ));

		Если (Лев(ИмяКлассаВрег, Длина) = ПрефиксИлиСуффикс1) или (Прав(ИмяКлассаВрег, Длина) = ПрефиксИлиСуффикс1) или (Длина = 0) Тогда
			Ошибка = "";
			Если ПустоеЗначение(набор.ПопыткаСоздатьОбъект(ИмяКласса, Ошибка)) = 1 Тогда
				Ошибка  =  "Создание класса '"+ИмяКласса+"' закончилось неудачей"+РазделительСтрок+
    	    		"	в тестировании класс не участвует"+РазделительСтрок+
					"   "+Ошибка;
				Сообщить(Ошибка, "!!!");
				Продолжить;
			КонецЕсли;
			Если набор.ЭтоТест(ИмяКласса) = 1 Тогда
				Список.ДобавитьЗначение(ИмяКласса);
			Иначе
	    	    Ошибка  =  "Класс   '"+ИмяКласса+"' не является потомком класса ТестовыйСлучай"+РазделительСтрок+
    	    		"	в тестировании класс не участвует";
				Сообщить(Ошибка, "!!!");
			КонецЕсли;
		КонецЕсли;
	конеццикла;

	возврат Список;
конецфункции

Процедура ЗагрузитьДополнительныеКлассы(КаталогФайловОписаний) Экспорт
	Состояние("Загружаю доп.классы из каталога "+КаталогФайловОписаний);
	
	Если Прав(КаталогФайловОписаний, 1) <> "\" Тогда
		КаталогФайловОписаний = КаталогФайловОписаний + "\";
	КонецЕсли;
	
	МетаИнфо = СоздатьОбъект("MetaInfoClasses");
	
	ПутьКФайлуОписания = ФС.НайтиПервыйФайл(КаталогФайловОписаний+"*.prm");
	Пока ПутьКФайлуОписания <> "" Цикл
		Сообщить("Загружаю доп.классы из файла "+КаталогФайловОписаний+ПутьКФайлуОписания);
		Попытка
			МетаИнфо.ЗагрузитьОписаниеКлассов(КаталогФайловОписаний+ПутьКФайлуОписания);
		Исключение
			Сообщить("Неудача при загрузке классов из каталога "+КаталогФайловОписаний, "!!!");
			Сообщить(ОписаниеОшибки(), "!!!");
			Сообщить("  ");
		КонецПопытки;
		
		ПутьКФайлуОписания = ФС.НайтиСледующийФайл();
	КонецЦикла;

	Состояние("");
