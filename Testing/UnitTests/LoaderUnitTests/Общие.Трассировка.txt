//******************************************************************************
// СообщениеТрассировки(Сообщение, ТипСообщения, ИзменениеУровняСообщения, стрПараметры, ...)
//
// Параметры: 
//  Сообщение - текст сообщения трассировки
//  ТипСообщения - число 0 - обычное сообщение (просто комментарий), 
//                                           1 - "красное" сообщение (фатальная ошибка)
//                                           -1 - "синее" сообщение (предупреждение)
//  ИзменениеУровняСообщения - число "1" - для увеличения уровня трассировки (сообщение имеет "вложенный" характер)
//           	                  Текущее сообщение и все последующие будут "на уровень больше" (правее)
//                             число "-1" - для уменьшения уровня трассировки
//  !!!  !!! !!!   	               Текущее сообщение и все последующие сообщения будут "на уровень меньше" (левее) !!! 
//
// стрПараметры - обозначения неопределенного числа параметров (например, "{Форма, 1,2, Контрол}" )
// может быть пара параметров
//  %%1, %%2 и т.п. - параметр, подставляемый в текст сообщения
//  сразу за ним - форматная строка, используемая для соответствующего параметра (ф.Формат)
//
// а также:
//  Форма -     необязательный параметр - форма, которую нужно открыть 
//              для обработки сообщения
//  Объект -    необязательный параметр - объект, который нужно активизировать (для форм списка) 
//				для обработки сообщения
//  Контрол -   необязательный параметр - контрол формы, который нужно активизировать для обработки сообщения
//
// Описание:
//  Вводит очередное сообщение трассировки, если параметр _Трассировка не является 
//  подходящим списком значений - выводится сообщение в окно Сообщений
//  При выводе сообщения производится подстановка подстроки %%1, %%2 и т.д. вместо соответствующих 
//  параметров в соответствующем формате.
//
// Например:
//           Трассировка.СообщениеТрассировки("Число: %%1", 0, 0, "{}", 340.23, "Ч(0)9.3")
//           приведет к сообщению - "Число 00340.230"
//
//           Трассировка.СообщениеТрассировки("Число: %%1", 1, 1, "{1,2,Форма}", 340.23, "Ч(0)9.3", Док)
//           приведет к красному сообщению со сдвигом вправо - "Число 00340.230" с расшифровкой
//

//Процедура СообщениеТрассировки(Сообщение, ТипСообщения, ИзменениеУровняСообщения, стрПараметры) Экспорт
Процедура СообщениеТрассировки(Сообщение, стрПараметры) Экспорт
	//Если (РежимТрассировки = 1) или (ТипСообщения<>0) Тогда 
	Если РежимТрассировки = 0 Тогда
		Возврат;
	КонецЕсли;                        

	Если ПустоеЗначение(стрПараметры) = 0 Тогда
		//Попытка                   
			спПараметры = Сам().ПолучитьСписокПараметров("СообщениеТрассировки");
		//Исключение                                   
		//	спПараметры = СоздатьОбъект("СписокЗначений");
		//КонецПопытки;
	Иначе
		спПараметры = СоздатьОбъект("СписокЗначений");
	КонецЕсли;
               
	Если спПараметры.РазмерСписка()=0 Тогда
		ИзменениеУровняСообщения = 0;
		Форма = "";
		Объект = "";
		Контрол = "";  
		ТипСообщения = "";
	Иначе
		Форма = спПараметры.Получить("Форма");
		Объект = спПараметры.Получить("Объект");
		Контрол = спПараметры.Получить("Контрол");  
		ТипСообщения = ПоУмолчанию(спПараметры.Получить("ТипСообщения"), спПараметры.Получить("Тип"));
		ТипСообщения = ПоУмолчанию(ТипСообщения, 0);
		Если НЕ((РежимТрассировки = 1) или (ТипСообщения<>0)) Тогда
			Возврат;
		КонецЕсли;    
		
		ИзменениеУровняСообщения = спПараметры.Получить("ИзменениеУровняСообщения");
		Если ПустоеЗначение(ИзменениеУровняСообщения)=1 Тогда
			ИзменениеУровняСообщения = спПараметры.Получить("ИзменениеУровня");
			Если ПустоеЗначение(ИзменениеУровняСообщения)=1 Тогда
				ИзменениеУровняСообщения = 0;
			КонецЕсли;
		КонецЕсли;
		//ИзменениеУровняСообщения = ?(ПустоеЗначение(ИзменениеУровняСообщения)=1, 0, 1);
		
		СтрокаИсключений = "Тип,ТипСообщения,ИзменениеУровняСообщения,ИзменениеУровня,Форма,Объект,Контрол,";
		ИндексНеопределенногоПараметра = 0;
		Для Индекс=1 По спПараметры.РазмерСписка() Цикл     
			НазваниеПараметра = "";
			Значение = спПараметры.ПолучитьЗначение(Индекс, НазваниеПараметра);
			Если НазваниеПараметра <> "" Тогда
				Если Найти(СтрокаИсключений, НазваниеПараметра+",")>0 Тогда
					Продолжить;
				КонецЕсли;        
			КонецЕсли;
			Индекс = Индекс + 1;
			ИндексНеопределенногоПараметра = ИндексНеопределенногоПараметра + 1;
			Если Индекс <= спПараметры.РазмерСписка() Тогда
				ФорматЗначения = спПараметры.ПолучитьЗначение(Индекс);
			Иначе
				ФорматЗначения = "";
			КонецЕсли;
			Если ПустоеЗначение(ФорматЗначения)=0 Тогда
		      	Значение = "" + Формат(Значение, ФорматЗначения);
		      Иначе
		      	Значение = "" + Значение;
		      КонецЕсли;
			Сообщение = СтрЗаменить(Сообщение, "%%"+СокрЛП(ИндексНеопределенногоПараметра), Значение);
		КонецЦикла;
	КонецЕсли;

      ТекущийУровень = _Трассировка.ПолучитьЗначение(1);
// Артур -- !!!Внимание !! Изменил поведение
      //Если ИзменениеУровняСообщения>0 Тогда
      //      ТекущийУровень = ТекущийУровень+1;
      //      _Трассировка.УстановитьЗначение(1,ТекущийУровень);
      //КонецЕсли;
      ТекущийУровень = ТекущийУровень+ИзменениеУровняСообщения;
      _Трассировка.УстановитьЗначение(1,ТекущийУровень);
// завершение		  

      ТекстСообщения = "";

      Для Сч=1 По ТекущийУровень Цикл
            ТекстСообщения = ТекстСообщения+"    ";
      КонецЦикла;

      ТекстСообщения = ТекстСообщения+Сообщение;
      Если ИзменениеУровняСообщения<0 Тогда                                  
            ТекущийУровень = Макс(ТекущийУровень+ИзменениеУровняСообщения,0);
            _Трассировка.УстановитьЗначение(1,ТекущийУровень);
      КонецЕсли;

      ТЗ = _Трассировка.ПолучитьЗначение(2);
      ТЗ.НоваяСтрока();
      ТЗ.ТипСообщения = ТипСообщения;
      ТЗ.ТекстСообщения = ТекстСообщения;
      ТЗ.Форма = Форма;
      ТЗ.Объект = Объект;
      ТЗ.Контрол = Контрол;
КонецПроцедуры    // СообщениеТрассировки()   

//------------------------------------ ------------------------------------ 
Процедура Сообщить(Сообщение, Маркер) Экспорт
	Если Найти(Маркер, "!") > 0 Тогда                    
		//Тип = ?(Найти(Маркер, "!!!") > 0, 3, ?(Найти(Маркер, "!!") > 0, 2, 1));
		Сам().СообщениеТрассировки(Сообщение, "{Тип}", 1);
	ИначеЕсли Врег(Маркер)="I" Тогда
		Сам().СообщениеТрассировки(Сообщение, "{Тип}", -1);
	Иначе
		Сам().СообщениеТрассировки(Сообщение, "");
	КонецЕсли;
КонецПроцедуры // Сообщить
