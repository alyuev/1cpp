		ОБ = СоздатьОбъект("УправлениеНастройками");
	Исключение
		Возврат;
	КонецПопытки;
	
	Флаг = ОБ.Получить(Наименование);
	Если Флаг <> Значение Тогда
		Если Значение = 0 Тогда
			Сам().ПропуститьТест(, "Настройка <"+Наименование+"> включена, а должна быть выключена");
		Иначе
			Сам().ПропуститьТест(, "Настройка <"+Наименование+"> выключена, а должна быть включена");
		КонецЕсли;
	КонецЕсли;
КонецПроцедуры	// ПроверитьНастройку

Процедура ПроверитьТипЗначения(ЧтоПроверяем, стрИскомыйТипЗначения) Экспорт
	Если ТипЗначенияСтр(ЧтоПроверяем) <> стрИскомыйТипЗначения Тогда
		ВМ = СоздатьОбъект("ВыполняемыйМодуль");
		СтрокаИсключения = "Тип <"+ТипЗначенияСтр(ЧтоПроверяем)+"> не равен типу <"+стрИскомыйТипЗначения+">, а хотели, чтоб было равно.";
		ВМ.ВыброситьИскл(СтрокаИсключения, СтрокаИсключения);
	КонецЕсли;
КонецПроцедуры	// ПроверитьТипЗначения

Процедура ПроверитьИсключениеНаЧтениеАтрибута(Объект, ИмяАтрибута) Экспорт
	Перем ТекстМодуля;
	
	ТекстМодуля = "Перем лЗначение;"+РазделительСтрок;
	
	ТекстМодуля = ТекстМодуля + "
	|Функция Получить(Объект) Экспорт
	|	Возврат Объект."+ИмяАтрибута+";
	|КонецФункции
	|";
	
	ВМ = СоздатьОбъект("ВыполняемыйМодуль");
	ВМ.УстановитьМодуль(ТекстМодуля);

	Попытка                   
		ВМ.КомпилироватьИВыполнитьМодуль();		
		лРез = ВМ.Получить(Объект);
	Исключение
		Возврат;
	КонецПопытки;
	
	СтрокаИсключения = "Хотели исключения, а его не было.";
	Сам().ВыброситьИскл(СтрокаИсключения, СтрокаИсключения);

КонецПроцедуры
